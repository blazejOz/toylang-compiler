Statement      ::= VarDeclrStmt 
                 | AssignStmt 
                 | PrintStmt

VarDeclrStmt   ::= "int" IDENTIFIER [ "=" Expr ] ";"
AssignStmt     ::= IDENTIFIER "=" Expr ";"
PrintStmt      ::= "print" "(" Expr ")" ";"

Expr           ::= Term { ("+" | "-") Term }
Term           ::= Primary { ("*" | "/") Primary }

Primary        ::= INTEGER 
                 | IDENTIFIER 
                 | "(" Expr ")"
